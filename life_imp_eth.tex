pref.eth<-function(v){
  A<-rownrm(table(life[,c("eth",v)]))
  print(dim(A))
  eth<-row.names(A)
  
  mv <- colSums(A)/6.0
  A0 <- matrix( 0, nrow=6, ncol=4)
  for (r in 1:6){ A0[r,]<-mv }
  S <- A - A0
  explained.var<-rep(0,6)
  for (s in 1:6){
    explained.var[s]<-sum(S[s,]^2)/sum(A[s,]^2)
  }
  data.frame(eth=eth,explained=explained.var*100)
}


\end{verbatim}
